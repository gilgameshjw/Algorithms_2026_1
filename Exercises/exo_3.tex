\begin{problem}[Fibonacci Super Fast!]
\begin{enumerate}
    \item compute Fibonacci with the relation: 
    \[
            \begin{bmatrix}
            F_{n+1} \\
            F_{n}
            \end{bmatrix}
            =
            \begin{bmatrix}
            1 & 1 \\
            1 & 0
            \end{bmatrix}^{\!n}
            \begin{bmatrix}
            1 \\
            0
            \end{bmatrix}
            \]
    \item This can also be expressed as:
    \[\begin{bmatrix}
    
            F_{n+1} \\
            F_{n}
            \end{bmatrix}
            =
            \left(\begin{bmatrix}
            2 & 1 \\
            1 & 1
            \end{bmatrix}\right)^{\!n/2}
            \begin{bmatrix}
            1 \\
            0
            \end{bmatrix}
            ,\]
        use Master Theorem to discuss the complexity of this decomposition and show, explain why time complexity is then $\mathrm{log}_2(n)$. 
\end{enumerate}
\end{problem}
